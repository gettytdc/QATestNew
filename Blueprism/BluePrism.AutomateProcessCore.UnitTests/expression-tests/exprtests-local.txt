#
#	exprtests-local.txt
#
#	List of expressions and their expected outputs used by the nunit test
#   defined in ExpressionTests.TestLocalExpressions()
#
#	File format:
#
#	  Blank lines, and lines beginning with # are ignored
#     Lines with a culture name within chevrons in an Xml style will
#       specify that a particular culture should apply to the subsequent tests.
#     Lines with a similar 'closing tag' will return the culture to
#       the default for the current user.
#	  All other lines are expression<tab>result<tab>resulttype
#     ... where 'expression' should be in the local form according to the currently
#     configured culture, and 'result' should be in normal form - ie. encoded form
#
# -----------------------------------------------------------------------------------
<en-GB>

# Date should be treated as 2014-02-01 in en-GB culture
AddDays("1/2/2014", 5)	2014/02/06	date

# Dot is decimal separator
1.0 + 2.5 - 3.25 + 1.75	2	number
1.3 * 3	3.9	number

</en-GB>
# -----------------------------------------------------------------------------------
<fr-FR>

# Date should work just like en-GB
AddDays("1/2/2014", 5)	2014/02/06	date

# Comma is decimal separator
1,0 + 2,5 - 3,25 + 1,75	2	number
1,3*3	3.9	number
4,5-Sqrt(9)	1.5	number

</fr-FR>
# -----------------------------------------------------------------------------------
<en-US>

# Date should be treated as 2014-01-02 in en-US culture
AddDays("1/2/2014", 5)	2014/01/07	date

# Number should work just like en-GB
1.3 * 3	3.9	number

</en-US>
# -----------------------------------------------------------------------------------
