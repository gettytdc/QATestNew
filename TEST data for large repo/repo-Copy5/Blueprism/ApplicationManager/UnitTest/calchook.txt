
#Start calclator in hooked mode...
startapplication path="calc.exe" hook=true

#Wait for the calculator to appear...
wait timeout=10000 {checkwindow windowtext=Calculator}=True
$AbortIf RESULT:0

#Get the text from the caption of the 'Hex' radio button
gettext windowtext="Hex" ClassName="Button" CtrlID="0" Visible="True" Enabled="True" pCtrlID="0" StartX="16" StartY="2" EndX="43" EndY="14"
$AbortIfNot RESULT:Hex
